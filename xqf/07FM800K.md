#!/usr/bin/python3
import sys

def main(argc, argv):
    A1        = 67
    A2        =  5
    A3        =  5
    A4        =  3
    B1        = 13.7
    B2        = 1.2
    B3a       = 3.2
    B3b       = 3.2
    B4        = 20

    o_sum     = A1 + A2 + A3 + A4
    i_sum_lia = B1 + B2 + B3a
    i_sum_lib = i_sum_lia + B3b
    i_sum     = i_sum_lib + B4
    x         = o_sum - i_sum

    C1ccq     = 22 # Final solution

    print(">>> A1        = %d" % A1)
    print(">>> A2        = %d" % A2)
    print(">>> A3        = %d" % A3)
    print(">>> A4        = %d" % A4)
    print(">>> o_sum     = A1 + A2 + A3 + A4 #= %d" % o_sum)
    print(">>> B1        = %.1f" % B1)
    print(">>> B2        = %.1f" % B2)
    print(">>> B3a       = %.1f" % B3a)
    print(">>> B3b       = %.1f" % B3b)
    print(">>> B4        = %d"   % B4)
    print(">>> i_sum_lia = B1 + B2 + B3a     #= %.1f" % i_sum_lia)
    print(">>> i_sum_lib = i_sum_lia + B3b   #= %.1f" % i_sum_lib)
    print(">>> i_sum     = i_sum_lib + B4    #= %.1f" % i_sum)
    print(">>> x         = o_sum - i_sum     #= %.1f ~= %d" %(x, round(x)))

    X = round(x)

         (A1, A1p, B4, C1ccq, kaiba, round(x628)))

    return 0

if __name__ == '__main__':
    sys.exit(main(len(sys.argv), sys.argv))
