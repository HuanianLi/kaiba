#!/usr/bin/python3
import sys

def main(argc, argv):
    A1        = 67
    A2        =  5
    A3        =  5
    A4        =  3
    B1        = 13.7
    B2        = 1.2
    B3a       = 3.2
    B3b       = 3.2
    B4        = 20

    o_sum     = A1 + A2 + A3 + A4
    i_sum_lia = B1 + B2 + B3a
    i_sum_lib = i_sum_lia + B3b
    i_sum     = i_sum_lib + B4
    x         = o_sum - i_sum

    C1lin     = 30 # solution 1
    C1ccq     = 20 # solution 2

    print(">>> A1        = %d" % A1)
    print(">>> A2        = %d" % A2)
    print(">>> A3        = %d" % A3)
    print(">>> A4        = %d" % A4)
    print(">>> o_sum     = A1 + A2 + A3 + A4 #= %d" % o_sum)
    print(">>> B1        = %.1f" % B1)
    print(">>> B2        = %.1f" % B2)
    print(">>> B3a       = %.1f" % B3a)
    print(">>> B3b       = %.1f" % B3b)
    print(">>> B4        = %d"   % B4)
    print(">>> i_sum_lia = B1 + B2 + B3a     #= %.1f" % i_sum_lia)
    print(">>> i_sum_lib = i_sum_lia + B3b   #= %.1f" % i_sum_lib)
    print(">>> i_sum     = i_sum_lib + B4    #= %.1f" % i_sum)
    print(">>> x         = o_sum - i_sum     #= %.1f ~= %d" %(x, round(x)))

    X = round(x)
    print(">>> #01 ? %d  = %d (Lin) + (%d) (Oth)" % (X, C1lin, X - C1lin))
    print(">>> OR")
    print(">>> #02 ? %d  = %d (Ccq) + (%d) (Oth)" % (X, C1ccq, X - C1ccq))

    kaiba = B1 + B2 + B3a
    x610  = A1 - (B4 + C1ccq + kaiba)
    print(">>> # [06/09] %d      - %d(lv) - %d(ccq) - %.1f(kaiba) = %d" % \
         (A1, B4, C1ccq, kaiba, round(x610)))

    A1p   = A2 + A3 + A4
    kaiba = B1 + B2 + B3a + B3b
    x628  = (A1 + A1p) - (B4 + C1ccq + kaiba)
    print(">>> # [06/27] %d + %d - %d(lv) - %d(ccq) - %.1f(kaiba) = %d" % \
         (A1, A1p, B4, C1ccq, kaiba, round(x628)))

    return 0

if __name__ == '__main__':
    sys.exit(main(len(sys.argv), sys.argv))
